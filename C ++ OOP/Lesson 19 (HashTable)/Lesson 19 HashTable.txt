            {

                table[hash].push_back(data); 
            }

                for (int value : table[hash])
                {

                }
                return false;
            }

            void print()
            {
                for (int i = 0; i <5; ++i)
                {
                    cout << i << " | ";
                    for (int val : table[i])

                    cout << endl;
                }
            }
    };
